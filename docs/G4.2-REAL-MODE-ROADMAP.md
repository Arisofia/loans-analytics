## G4.2.3: Harden REAL vs MOCK Switching in Orchestrator

**Objective:** Centralize and audit how orchestrator chooses REAL vs MOCK mode.

### Create Config Module

File: `python/multi_agent/config_historical.py`

```python
"""Config module for historical KPI provider (REAL vs MOCK)."""

import os
from python.multi_agent.historical_context import HistoricalContextProvider
from python.multi_agent.historical_backend_supabase import SupabaseHistoricalBackend


def build_historical_context_provider(
    cache_ttl_seconds: int = 60,
    force_mode: str | None = None,
) -> HistoricalContextProvider:
    """
    Build a HistoricalContextProvider based on environment.

    Args:
        cache_ttl_seconds: Cache TTL for KPI queries.
        force_mode: Override mode selection ("MOCK" or "REAL"). For testing only.

    Returns:
        HistoricalContextProvider configured for MOCK or REAL.

    Environment Variables:
        HISTORICAL_CONTEXT_MODE: "MOCK" (default) or "REAL"
        SUPABASE_URL: Required if mode is REAL
        SUPABASE_ANON_KEY: Required if mode is REAL
    """
    mode = (force_mode or os.getenv("HISTORICAL_CONTEXT_MODE", "MOCK")).upper()

    if mode not in ("MOCK", "REAL"):
        raise ValueError(f"Invalid mode: {mode}. Must be MOCK or REAL.")

    if mode == "REAL":
        # Validate Supabase credentials
        url = os.getenv("SUPABASE_URL")
        key = os.getenv("SUPABASE_ANON_KEY")
        if not url or not key:
            raise ValueError(
                "REAL mode requires SUPABASE_URL and SUPABASE_ANON_KEY."
            )

        backend = SupabaseHistoricalBackend()
        provider = HistoricalContextProvider(
            cache_ttl_seconds=cache_ttl_seconds,
            mode="REAL",
            backend=backend,
        )
    else:
        provider = HistoricalContextProvider(cache_ttl_seconds=cache_ttl_seconds)

    return provider


if __name__ == "__main__":
    # CLI test: verify mode selection
    provider = build_historical_context_provider()
    mode = os.getenv("HISTORICAL_CONTEXT_MODE", "MOCK")
    print(f"Built provider: mode={mode}")
```
